% Parametrized Frontpage layout command
\newcommand\frontpage[1][1]{
	\hypersetup{pageanchor=false}                   
	\ifnum#1<10
	\input{frontpages/frontpage0#1.tex}
	\else
	\input{frontpages/frontpage#1.tex}
	\fi
}


% Thesis indexes generation command
\newcommand\thesisindex[1][]{
	\hypersetup{pageanchor=true}                    
	\pagenumbering{roman}
	{
		\pagestyle{indexpage}
		\hypersetup{linkcolor=black}
		\tableofcontents
		
		\newpage
		\listoffigures
		\listoftables
		
		\newpage
		\printnoidxglossary[type=\acronymtype,title=Acronyms]
	}
	\newpage\null\newpage
	\clearpage
}

% Quotation environment for courtesy pages
\NewDocumentEnvironment{thesisquoteenv}{m +b}{%
	\vspace*{\fill}
	\par\addvspace{\topsep}
	\begin{samepage}
		\raggedleft\guillemotright#2\guillemotleft
		\par\nopagebreak
		{\footnotesize\raggedleft\textbf{#1}\par}
	\end{samepage}%
	\par\addvspace{\topsep}
	\vspace*{\fill}
}{}

% Optional quotation page generation command
\newcommand\thesisquotepage[1][]{
	\thispagestyle{empty}
	\begin{thesisquoteenv}{\thesisquoteauthor}
		\thesisquote
	\end{thesisquoteenv}
	\newpage\null\thispagestyle{empty}\newpage
}

% Parametrized Acknowledgments page layout command
\newcommand\thesisacknowledgments[1][]{
	\pagestyle{indexpage}
	\input{sections/acknowledgments.tex}     	% Load acknowledgments content
	\clearpage
	\newpage\null\thispagestyle{empty}\newpage	% Blank transition page
}

% Parametrized Abstract page layout command (resets numbering to Arabic)
\newcommand\thesisabstract[1][]{
	\clearpage
	\pagenumbering{arabic}
	\setcounter{page}{1}
	\pagestyle{abastractpage}
	\phantomsection
	\addcontentsline{toc}{section}{Abstract} 
	\input{sections/abstract.tex}           	% Load abstract content
	\clearpage
	\pagestyle{documentpage}               		% Set standard page styling
}

% Parametrized Bibliography page layout command
\newcommand\thesisbibliography[1][]{
	\clearpage
	\newpage
	\pagestyle{abastractpage}
	\printbibliography[heading=bibintoc]     	% Generate bibliography index
}
